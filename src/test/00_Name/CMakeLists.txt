GET_DIR_NAME(DIR_NAME)
SET(TARGET_NAME "${TARGET_PREFIX}${DIR_NAME}")
FILE(GLOB SOURCES
        "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp"
        "${CMAKE_CURRENT_SOURCE_DIR}/*.h"
        "${CMAKE_CURRENT_SOURCE_DIR}/*.inl"
)

SET(LIBS "")
ADD_TARGET(MODE ${MODE} NAME ${TARGET_NAME} SOURCES ${SOURCES} LIBS ${LIBS})
